const int NUM_LIMBS = 4;  ;; Assuming 4 limbs for 256-bit numbers (64-bit limbs)

;; ===========================================================
;; 3. Helper Functions for Multi-Limb Arithmetic
;; ===========================================================

;; Subtract two multi-limb numbers with modular reduction
(int, int, int, int) my_sub_mod_limbs(int a0, int a1, int a2, int a3, int b0, int b1, int b2, int b3) inline {
    ;; Perform subtraction on the first limb
    int diff0 = a0 - b0;
    int borrow0 = (diff0 < 0) ? 1 : 0;

    ;; If the result is negative, add the modulus (Goldilocks prime)
    if (borrow0 != 0) {
        diff0 += P0_0;
    }

    ;; Since higher limbs are zero, return zero for them
    return (diff0 & ((1 << LIMB_SIZE) - 1), 0, 0, 0);
}

(int, int, int, int) my_add_mod_limbs(int a0, int a1, int a2, int a3, int b0, int b1, int b2, int b3) inline {
    int c0 = a0 + b0;
    int carry0 = c0 >> LIMB_SIZE;
    c0 &= ((1 << LIMB_SIZE) - 1);

    ;; We only care about the first limb since the prime fits within 64 bits.
    int need_reduce = (c0 >= P0_0);
    
    if (need_reduce != 0) {
        ;; Reduce by Goldilocks prime
        c0 -= P0_0;
    }

    return (c0, 0, 0, 0);  ;; Higher limbs are zero
}


;; Multiply two multi-limb numbers with modular reduction
(int, int, int, int) my_mul_mod_limbs(int a0, int a1, int a2, int a3, int b0, int b1, int b2, int b3) inline {
    ;; Perform full multiplication on the first limb
    int c0 = a0 * b0;

    ;; Since the prime fits in one limb, we only need to reduce on the first limb
    int r0 = c0 & ((1 << LIMB_SIZE) - 1);
    int carry = c0 >> LIMB_SIZE;

    ;; If there's overflow, add it to the result
    r0 += carry;

    ;; If r0 exceeds the prime, reduce it by subtracting the prime
    if (r0 >= P0_0) {
        r0 -= P0_0;
    }

    ;; Since the Goldilocks prime is small, no need for higher limbs
    return (r0, 0, 0, 0);
}


(int, int, int, int) my_sbox_limbs(int x0, int x1, int x2, int x3) inline {
    ;; x^2
    (int x2_0, int x2_1, int x2_2, int x2_3) = my_mul_mod_limbs(x0, x1, x2, x3, x0, x1, x2, x3);

    ;; x^4
    (int x4_0, int x4_1, int x4_2, int x4_3) = my_mul_mod_limbs(x2_0, x2_1, x2_2, x2_3, x2_0, x2_1, x2_2, x2_3);

    ;; x^5
    (int x5_0, int x5_1, int x5_2, int x5_3) = my_mul_mod_limbs(x4_0, x4_1, x4_2, x4_3, x0, x1, x2, x3);

    return (x5_0, x5_1, x5_2, x5_3);
}

;; Applies the MDS matrix to the current state
(int, int, int, int) my_apply_mds(
    int s00, int s01, int s02, int s03,  ;; State 0
    int s10, int s11, int s12, int s13,  ;; State 1
    int s20, int s21, int s22, int s23,  ;; State 2
    int s30, int s31, int s32, int s33   ;; State 3
) inline_ref {
    
    ;; Compute each new state element by performing the dot product
    ;; of each MDS matrix row with the current state vector.

    ;; New State 0 = M00*s00 + M01*s10 + M02*s20 + M03*s30
    int mul0_0 = my_mul_mod_limbs(M00, 0, 0, 0, s00, s01, s02, s03);
    int mul1_0 = my_mul_mod_limbs(M01, 0, 0, 0, s10, s11, s12, s13);
    int mul2_0 = my_mul_mod_limbs(M02, 0, 0, 0, s20, s21, s22, s23);
    int mul3_0 = my_mul_mod_limbs(M03, 0, 0, 0, s30, s31, s32, s33);
    int new_s00 = my_add_mod_limbs(mul0_0, 0, 0, 0, mul1_0, 0, 0, 0);
    new_s00 = my_add_mod_limbs(new_s00, 0, 0, 0, mul2_0, 0, 0, 0);
    new_s00 = my_add_mod_limbs(new_s00, 0, 0, 0, mul3_0, 0, 0, 0);

    ;; New State 1 = M10*s00 + M11*s10 + M12*s20 + M13*s30
    int mul0_1 = my_mul_mod_limbs(M10, 0, 0, 0, s00, s01, s02, s03);
    int mul1_1 = my_mul_mod_limbs(M11, 0, 0, 0, s10, s11, s12, s13);
    int mul2_1 = my_mul_mod_limbs(M12, 0, 0, 0, s20, s21, s22, s23);
    int mul3_1 = my_mul_mod_limbs(M13, 0, 0, 0, s30, s31, s32, s33);
    int new_s10 = my_add_mod_limbs(mul0_1, 0, 0, 0, mul1_1, 0, 0, 0);
    new_s10 = my_add_mod_limbs(new_s10, 0, 0, 0, mul2_1, 0, 0, 0);
    new_s10 = my_add_mod_limbs(new_s10, 0, 0, 0, mul3_1, 0, 0, 0);

    ;; New State 2 = M20*s00 + M21*s10 + M22*s20 + M23*s30
    int mul0_2 = my_mul_mod_limbs(M20, 0, 0, 0, s00, s01, s02, s03);
    int mul1_2 = my_mul_mod_limbs(M21, 0, 0, 0, s10, s11, s12, s13);
    int mul2_2 = my_mul_mod_limbs(M22, 0, 0, 0, s20, s21, s22, s23);
    int mul3_2 = my_mul_mod_limbs(M23, 0, 0, 0, s30, s31, s32, s33);
    int new_s20 = my_add_mod_limbs(mul0_2, 0, 0, 0, mul1_2, 0, 0, 0);
    new_s20 = my_add_mod_limbs(new_s20, 0, 0, 0, mul2_2, 0, 0, 0);
    new_s20 = my_add_mod_limbs(new_s20, 0, 0, 0, mul3_2, 0, 0, 0);

    ;; New State 3 = M30*s00 + M31*s10 + M32*s20 + M33*s30
    int mul0_3 = my_mul_mod_limbs(M30, 0, 0, 0, s00, s01, s02, s03);
    int mul1_3 = my_mul_mod_limbs(M31, 0, 0, 0, s10, s11, s12, s13);
    int mul2_3 = my_mul_mod_limbs(M32, 0, 0, 0, s20, s21, s22, s23);
    int mul3_3 = my_mul_mod_limbs(M33, 0, 0, 0, s30, s31, s32, s33);
    int new_s30 = my_add_mod_limbs(mul0_3, 0, 0, 0, mul1_3, 0, 0, 0);
    new_s30 = my_add_mod_limbs(new_s30, 0, 0, 0, mul2_3, 0, 0, 0);
    new_s30 = my_add_mod_limbs(new_s30, 0, 0, 0, mul3_3, 0, 0, 0);

    ;; ===========================================================
    ;; Return the New State After MDS Multiplication
    ;; ===========================================================
    return (
        new_s00, 0, 0, 0,  ;; s00 updated, higher limbs set to 0
        new_s10, 0, 0, 0,  ;; s10 updated, higher limbs set to 0
        new_s20, 0, 0, 0,  ;; s20 updated, higher limbs set to 0
        new_s30, 0, 0, 0   ;; s30 updated, higher limbs set to 0
    );
}


;; ===========================================================
;; 6. Poseidon2 Permutation
;; ===========================================================
(int, int, int, int, 
 int, int, int, int, 
 int, int, int, int, 
 int, int, int, int) my_poseidon2_permutation(
    int s00, int s01, int s02, int s03,  ;; State 0
    int s10, int s11, int s12, int s13,  ;; State 1
    int s20, int s21, int s22, int s23,  ;; State 2
    int s30, int s31, int s32, int s33   ;; State 3
) 
{
    ;; Initialize local state variables
    int local_s00 = s00; int local_s01 = s01; int local_s02 = s02; int local_s03 = s03;
    int local_s10 = s10; int local_s11 = s11; int local_s12 = s12; int local_s13 = s13;
    int local_s20 = s20; int local_s21 = s21; int local_s22 = s22; int local_s23 = s23;
    int local_s30 = s30; int local_s31 = s31; int local_s32 = s32; int local_s33 = s33;

    int i = 0;
    while (i < TOTAL_ROUNDS) {
        ;; ===========================================================
        ;; Add Round Constants to Each State Element
        ;; ===========================================================
        ;; State 0
        (int rc0, int rc1, int rc2, int rc3) = my_get_round_constant(i * T + 0);
        (int ns00, int ns01, int ns02, int ns03) = my_add_mod_limbs(local_s00, local_s01, local_s02, local_s03, rc0, rc1, rc2, rc3);
        local_s00 = ns00; local_s01 = ns01; local_s02 = ns02; local_s03 = ns03;

        ;; State 1
        (rc0, rc1, rc2, rc3) = my_get_round_constant(i * T + 1);
        (int ns10, int ns11, int ns12, int ns13) = my_add_mod_limbs(local_s10, local_s11, local_s12, local_s13, rc0, rc1, rc2, rc3);
        local_s10 = ns10; local_s11 = ns11; local_s12 = ns12; local_s13 = ns13;

        ;; State 2
        (rc0, rc1, rc2, rc3) = my_get_round_constant(i * T + 2);
        (int ns20, int ns21, int ns22, int ns23) = my_add_mod_limbs(local_s20, local_s21, local_s22, local_s23, rc0, rc1, rc2, rc3);
        local_s20 = ns20; local_s21 = ns21; local_s22 = ns22; local_s23 = ns23;

        ;; State 3
        (rc0, rc1, rc2, rc3) = my_get_round_constant(i * T + 3);
        (int ns30, int ns31, int ns32, int ns33) = my_add_mod_limbs(local_s30, local_s31, local_s32, local_s33, rc0, rc1, rc2, rc3);
        local_s30 = ns30; local_s31 = ns31; local_s32 = ns32; local_s33 = ns33;

        ;; ===========================================================
        ;; Apply S-Box Transformation
        ;; ===========================================================
        if ((i < (RF_FULL / 2)) | (i >= (RF_FULL / 2 + RP_PARTIAL))) {
            ;; Full S-Box: Apply to all state elements
            (local_s00, local_s01, local_s02, local_s03) = my_sbox_limbs(local_s00, local_s01, local_s02, local_s03);
            (local_s10, local_s11, local_s12, local_s13) = my_sbox_limbs(local_s10, local_s11, local_s12, local_s13);
            (local_s20, local_s21, local_s22, local_s23) = my_sbox_limbs(local_s20, local_s21, local_s22, local_s23);
            (local_s30, local_s31, local_s32, local_s33) = my_sbox_limbs(local_s30, local_s31, local_s32, local_s33);
        } else {
            ;; Partial S-Box: Apply only to the first state element
            (local_s00, local_s01, local_s02, local_s03) = my_sbox_limbs(local_s00, local_s01, local_s02, local_s03);
            ;; Other state elements remain unchanged
        }

        ;; ===========================================================
        ;; Apply Linear Layer (MDS Matrix Multiplication)
        ;; ===========================================================
        ;; Replace the flawed MDS multiplication code with a call to the helper function
        (local_s00, local_s10, local_s20, local_s30) = my_apply_mds(
            local_s00, local_s01, local_s02, local_s03,
            local_s10, local_s11, local_s12, local_s13,
            local_s20, local_s21, local_s22, local_s23,
            local_s30, local_s31, local_s32, local_s33
        );

        ;; Increment Round Counter
        i += 1;
    }

    ;; ===========================================================
    ;; Return the Updated State After All Rounds
    ;; ===========================================================
    return (
        local_s00, local_s01, local_s02, local_s03,
        local_s10, local_s11, local_s12, local_s13,
        local_s20, local_s21, local_s22, local_s23,
        local_s30, local_s31, local_s32, local_s33
    );
}

;; ===========================================================
;; 7. Poseidon2 Hash Function
;; ===========================================================

(int, int) my_poseidon2_hash(
    int in00, int in01, int in02, int in03,  ;; Input 0
    int in10, int in11, int in12, int in13   ;; Input 1
) {
    ;; Initialize state with inputs and capacity elements set to zero
    var (s00, s01, s02, s03,
         s10, s11, s12, s13,
         s20, s21, s22, s23,
         s30, s31, s32, s33) = my_poseidon2_permutation(
            in00, in01, in02, in03,
            in10, in11, in12, in13,
            0, 0, 0, 0,  ;; Capacity element 1
            0, 0, 0, 0   ;; Capacity element 2
    );

    ;; Return only two state elements (e.g., s00, s01) as the final hash
    return (s00, s01);
}

;; For off-chain retrieval or testing
(int, int) my_get_poseidon2_hash(
    int in00, int in01, int in02, int in03,
    int in10, int in11, int in12, int in13
) method_id {
    return my_poseidon2_hash(in00, in01, in02, in03, in10, in11, in12, in13);
}

;; ===========================================================
;; 8. Contract Entry Point
;; ===========================================================

() my_recv_internal(int msg_value, cell in_msg_cell, slice in_msg) impure {
    ;; Parse the incoming message
    var cs = in_msg_cell.begin_parse();
    var flags = cs~load_uint(4);

    ;; Ignore all bounced messages
    if (flags & 1) {
        return ();
    }

    ;; Extract the operation type from the incoming message
    if (in_msg.slice_empty?()) {
        return ();  ;; Exit if the message is empty
    }
    var op = in_msg~load_uint(32);

    ;; Handle different operations
    if (op == 1) {  ;; Compute Poseidon2 hash
        ;; Ensure there are enough bits for inputs
        if (in_msg.slice_bits() < (2 * NUM_LIMBS * LIMB_SIZE)) {
            throw(0x01);  ;; Invalid input length
        }

        ;; Extract the inputs
        var in00 = in_msg~load_uint(LIMB_SIZE);
        var in01 = in_msg~load_uint(LIMB_SIZE);
        var in02 = in_msg~load_uint(LIMB_SIZE);
        var in03 = in_msg~load_uint(LIMB_SIZE);

        var in10 = in_msg~load_uint(LIMB_SIZE);
        var in11 = in_msg~load_uint(LIMB_SIZE);
        var in12 = in_msg~load_uint(LIMB_SIZE);
        var in13 = in_msg~load_uint(LIMB_SIZE);

        ;; Compute the Poseidon2 hash
        var (hash00, hash01) = my_poseidon2_hash(
            in00, in01, in02, in03, in10, in11, in12, in13
        );

        ;; Construct the response message (only two hash values)
        var msg = begin_cell()
            .store_uint(0x10, 6)          ;; nobounce flag
            .store_slice(my_address())    ;; sender address
            .store_uint(0x076ef1ea, 32)   ;; operation code for outgoing message
            .store_uint(hash00, LIMB_SIZE)
            .store_uint(hash01, LIMB_SIZE)
            .end_cell();

        ;; Send the hash result back
        send_raw_message(msg, 64); ;; mode 64: carry all remaining value of the inbound message
    } else {
        throw(0xffff);  ;; Throw exception for unknown operations
    }
}

;; ===========================================================
;; 9. Main Function
;; ===========================================================

() main() {
    var in_msg_cell = get_data();  ;; Retrieve the incoming message cell
    var in_msg = in_msg_cell.begin_parse();  ;; Start parsing the incoming message
    my_recv_internal(0, in_msg_cell, in_msg);  ;; Call the recv_internal function with necessary parameters
}
